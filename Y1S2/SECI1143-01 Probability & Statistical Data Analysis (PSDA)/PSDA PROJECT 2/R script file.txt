Hypothesis Testing (2 samples Test):
# Load necessary libraries
library(readr)
library(readxl)
library(ggplot2)

# Load the data
heart1 <- read_excel("C:/Users/user/Downloads/heart1.xlsx")

# Separate the ages of males and females
male_ages <- heart1$age[heart1$sex == 1]
female_ages <- heart1$age[heart1$sex == 0]

# Calculate means
mean_male_age <- mean(male_ages, na.rm = TRUE)
mean_female_age <- mean(female_ages, na.rm = TRUE)

# Perform the two-sample t-test (assuming unequal variances)
t_test_result <- t.test(male_ages, female_ages, var.equal = FALSE)

# Print the results
print(t_test_result)

# Create a density plot with mean lines

Chi-squared Test of Independence:
# Load necessary library
library(readr)

# Load the dataset
df <- read_csv("heart1.csv")

# Convert columns to factors if they are not already
df$sex <- as.factor(df$sex)
df$cp <- as.factor(df$cp)

# Create a contingency table
contingency_table <- table(df$sex, df$cp)

# Perform Chi-Square Test of Independence
chi_square_test <- chisq.test(contingency_table)

# Display the results
chi_square_statistic <- chi_square_test$statistic
p_value <- chi_square_test$p.value
degrees_of_freedom <- chi_square_test$parameter
expected_frequencies <- chi_square_test$expected

print(paste("Chi-Square Statistic:", chi_square_statistic))
print(paste("P-value:", p_value))
print(paste("Degrees of Freedom:", degrees_of_freedom))
print("Expected Frequencies:")
print(expected_frequencies)
